Jenkins + Jasmine

Jenkins runs our unit tests on each check-in...

The JavaScript Task Runner

grunt-contrib-jasmine

Install with:


		$ npm install grunt-contrib-jasmine --save-dev

Enable in Gruntfile with:


		grunt.loadNpmTasks('grunt-contrib-jasmine');

Gruntfile.js


module.exports = function(grunt) {
	grunt.initConfig({
		jasmine: {
			src: 'src/**/*.js',
			options: {
				specs: 'spec/**/*.js'
			}
		}
	});
	grunt.loadNpmTasks('grunt-contrib');
	grunt.registerTask('test', ['jasmine']);
}

Run your tests


		$ grunt test

Gruntfile.js


module.exports = function(grunt) {
	grunt.initConfig({
		livereload: ..., 
		connect: ...,
		regarde: ...
	});

	grunt.loadNpmTasks('grunt-regarde');
	grunt.loadNpmTasks('grunt-contrib');

	grunt.registerTask('dev', ['livereload-start', 'connect', 'regarde']);
};

	server: {
		options: {
			port: 8558,
			base: 'public/',
			middleware: function(connect, options) {
				return [
					require('grunt-contrib/node_modules/grunt-contrib-livereload/lib/utils').livereloadSnippet, 
					connect.static(require('path').resolve(options.base))

}
